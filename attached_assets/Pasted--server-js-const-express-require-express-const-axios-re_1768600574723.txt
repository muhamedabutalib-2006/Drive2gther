// Extract folder ID from Google Drive URL
function extractFolderId(url) {
  const patterns = [
    /\/folders\/([a-zA-Z0-9-_]+)/,
    /id=([a-zA-Z0-9-_]+)/,
    /^([a-zA-Z0-9-_]+)$/
  ];
  
  for (const pattern of patterns) {
    const match = url.match(pattern);
    if (match) return match[1];
  }
  return null;
}

// Convert Google Drive file to streaming URL
function getDriveStreamUrl(fileId) {
  return `https://drive.google.com/file/d/${fileId}/preview`;
}

// Fetch video files from Google Drive folder
async function fetchVideos(folderId) {
  try {
    let allFiles = [];
    let pageToken = null;
    
    do {
      const url = `https://www.googleapis.com/drive/v3/files`;
      const params = {
        key: GOOGLE_API_KEY,
        q: `'${folderId}' in parents and (mimeType contains 'video/' or mimeType contains 'application/')`,
        fields: 'nextPageToken, files(id, name, mimeType, webViewLink, webContentLink)',
        pageSize: 1000, // Maximum allowed
        orderBy: 'name',
        supportsAllDrives: true,
        includeItemsFromAllDrives: true
      };
      
      if (pageToken) {
        params.pageToken = pageToken;
      }
      
      console.log('Fetching videos from folder:', folderId, 'Page token:', pageToken || 'first page');
      const response = await axios.get(url, { params });
      
      const files = response.data.files || [];
      allFiles = allFiles.concat(files);
      pageToken = response.data.nextPageToken;
      
      console.log('Fetched', files.length, 'files. Total so far:', allFiles.length);
    } while (pageToken);
    
    console.log('Total files found:', allFiles.length);
    
    // Add streaming URL to each video
    return allFiles.map(file => ({
      ...file,
      streamUrl: getDriveStreamUrl(file.id)
    }));
  } catch (error) {
    console.error('Error fetching videos:', error.response?.data || error.message);
    const errorMsg = error.response?.data?.error?.message || error.message;
    throw new Error(`Failed to fetch videos: ${errorMsg}`);
  }
}

// Create Watch2Gether room
async function createW2GRoom() {
  try {
    console.log('Creating W2G room...');
    const response = await axios.post(
      'https://api.w2g.tv/rooms/create.json',
      {
        w2g_api_key: W2G_API_KEY,
        share: 'https://www.youtube.com/watch?v=dQw4w9WgXcQ',
        bg_color: '#1a1a1a',
        bg_opacity: '90'
      }
    );
    console.log('W2G room created:', response.data.streamkey);
    return response.data.streamkey;
  } catch (error) {
    console.error('Error creating W2G room:', error.response?.data || error.message);
    const errorMsg = error.response?.data?.error || error.message;
    throw new Error(`Failed to create Watch2Gether room: ${errorMsg}`);
  }
}

// Update Watch2Gether room with new video
async function updateW2GRoom(roomId, videoUrl) {
  try {
    const response = await axios.post(
      `https://api.w2g.tv/rooms/${roomId}/sync_update`,
      {
        w2g_api_key: W2G_API_KEY,
        item_url: videoUrl
      }
    );
    return response.data;
  } catch (error) {
    console.error('Error updating W2G room:', error.response?.data || error.message);
    throw new Error(`Failed to update Watch2Gether room: ${errorMsg}`);
  }
}

// Initialize with folder
app.post('/api/init', async (req, res) => {
  try {
    const { folderUrl } = req.body;
    console.log('Received folder URL:', folderUrl);
    
    const folderId = extractFolderId(folderUrl);
    console.log('Extracted folder ID:', folderId);
    
    if (!folderId) {
      return res.status(400).json({ error: 'Invalid Google Drive folder URL' });
    }
    
    // Fetch videos
    const videos = await fetchVideos(folderId);
    console.log('Videos found:', videos.length);
    
    if (videos.length === 0) {
      return res.status(404).json({ 
        error: 'No videos found in folder. Make sure the folder is publicly accessible.' 
      });
    }
    
    // Sort videos by name with proper numeric sorting
    videos.sort((a, b) => {
      // Extract numbers from filenames (e.g., E01, Episode 1, 01, etc.)
      const getNumber = (name) => {
        const match = name.match(/(\d+)/);
        return match ? parseInt(match[1], 10) : 0;
      };
      
      const numA = getNumber(a.name);
      const numB = getNumber(b.name);
      
      // If both have numbers, sort by number
      if (numA !== numB) {
        return numA - numB;
      }
      
      // Otherwise, sort alphabetically
      return a.name.localeCompare(b.name, undefined, { numeric: true, sensitivity: 'base' });
    });
    
    // Create W2G room if needed
    if (!appState.roomId) {
      console.log('Creating new W2G room...');
      appState.roomId = await createW2GRoom();
    }
    
    // Update state
    appState.folderId = folderId;
    appState.episodes = videos;
    appState.currentEpisodeIndex = 0;
    
    // Load first episode
    if (videos.length > 0) {
      const firstVideo = videos[0];
      console.log('Loading first episode:', firstVideo.name, 'URL:', firstVideo.streamUrl);
      await updateW2GRoom(appState.roomId, firstVideo.streamUrl);
    }
    
    res.json({
      roomId: appState.roomId,
      episodes: videos.map(v => ({ id: v.id, name: v.name })),
      currentIndex: 0
    });
  } catch (error) {
    console.error('Error in /api/init:', error);
    res.status(500).json({ error: error.message });
  }
});

// Get current state
app.get('/api/state', (req, res) => {
  res.json({
    roomId: appState.roomId,
    episodes: appState.episodes.map(v => ({ id: v.id, name: v.name })),
    currentIndex: appState.currentEpisodeIndex
  });
});

// Play specific episode
app.post('/api/play', async (req, res) => {
  try {
    const { index } = req.body;
    
    if (!appState.roomId || appState.episodes.length === 0) {
      return res.status(400).json({ error: 'No playlist initialized' });
    }
    
    if (index < 0 || index >= appState.episodes.length) {
      return res.status(400).json({ error: 'Invalid episode index' });
    }
    
    appState.currentEpisodeIndex = index;
    const episode = appState.episodes[index];
    console.log('Playing episode:', episode.name, 'URL:', episode.streamUrl);
    await updateW2GRoom(appState.roomId, episode.streamUrl);
    
    res.json({
      currentIndex: appState.currentEpisodeIndex,
      episode: { id: episode.id, name: episode.name }
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Navigate episodes
app.post('/api/navigate', async (req, res) => {
  try {
    const { direction } = req.body;
    
    if (!appState.roomId || appState.episodes.length === 0) {
      return res.status(400).json({ error: 'No playlist initialized' });
    }
    
    let newIndex = appState.currentEpisodeIndex;
    if (direction === 'next') {
      newIndex = Math.min(newIndex + 1, appState.episodes.length - 1);
    } else if (direction === 'prev') {
      newIndex = Math.max(newIndex - 1, 0);
    }
    
    if (newIndex !== appState.currentEpisodeIndex) {
      appState.currentEpisodeIndex = newIndex;
      const episode = appState.episodes[newIndex];
      console.log('Navigating to episode:', episode.name, 'URL:', episode.streamUrl);
      await updateW2GRoom(appState.roomId, episode.streamUrl);
    }
    
    res.json({
      currentIndex: appState.currentEpisodeIndex,
      episode: appState.episodes[appState.currentEpisodeIndex]
    });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Create new room
app.post('/api/new-room', async (req, res) => {
  try {
    console.log('Creating new W2G room...');
    appState.roomId = await createW2GRoom();
    
    // If we have episodes loaded, load the current one
    if (appState.episodes.length > 0) {
      const episode = appState.episodes[appState.currentEpisodeIndex];
      await updateW2GRoom(appState.roomId, episode.streamUrl);
    }
    
    res.json({ roomId: appState.roomId });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
